<script>
const grids = [];
const xAxes = [];
const yAxes = [];
const series = [];
const titles = [];
let count = 0;
$.ajaxSettings.async = false;
$.getJSON('../24/data.json', function (data) {
  for (key in data) {
    var _data = [];
    for (item in data[key]) {
      var x = item;
      var y = data[key][item].total / data[key][item].num;
      _data.push([x, y.toFixed(2)]);
    }
    grids.push({
      show: true,
      borderWidth: 0,
      shadowColor: 'rgba(0, 0, 0, 0.3)',
      shadowBlur: 1,
      containLabel: true
    });
    xAxes.push({
      type: 'category',
      show: false,
      gridIndex: count,
      nameLocation: 'middle'
    });
    yAxes.push({
      type: 'value',
      show: false,
      gridIndex: count,
      nameLocation: 'left',
      boundaryGap: [0, '100%'],
    });
    series.push({
      name: key,
      type: 'line',
      xAxisIndex: count,
      yAxisIndex: count,
      data: _data,
      showSymbol: true,
      animationDuration: 1000,
      smooth: true,
      //areaStyle: {}
    });
    titles.push({
      textAlign: 'center',
      text: key,
      textStyle: {
        fontSize: 12,
        fontWeight: 'normal',
        color: '#ccc'
      },
    });
    count++;
  }
  var rowNumber = Math.ceil(Math.sqrt(count));
  $.each(grids, function (idx, grid) {
    grid.left = ((idx % rowNumber) / rowNumber) * 100 + 0.5 + '%';
    grid.top = (Math.floor(idx / rowNumber) / rowNumber) * 100 + 0.5 + '%';
    grid.width = (1 / rowNumber) * 100 - 1 + '%';
    grid.height = (1 / rowNumber) * 100 - 1 + '%';
    titles[idx].left = parseFloat(grid.left) + parseFloat(grid.width) / 2 + '%';
    titles[idx].top = parseFloat(grid.top) + parseFloat(grid.width) - 3.5 + '%';
  });
});
